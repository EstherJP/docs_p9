def each
    aux = @head

        yield aux.value
        aux = aux.next
    end
end

def extraer_head
    aux = @head
    @head = @head.next
    @head.prev = nil
    aux.next = nil

    @size -= 1
end

def insertar_nodo(v)
    aux = Node.new(v, nil, nil)
    if  @tail == nil
        @head = aux
        @tail = aux

    else 
        @tail.next = aux
        aux.prev = @tail
        @tail = aux
    end
    @size += 1
end

def to_s
    aux = @head
    a = '('

    while aux!= nil
        if aux == tail
            a << "#{aux.value}"
        else
            a << "#{aux.value}"
            a << ", "
        end
        aux = aux.next
    end
    a << ')'
    a
end
