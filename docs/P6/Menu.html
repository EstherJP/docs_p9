def estimar_ut
    ut_al = @alimentos.collect{|x| x.tr}
    gr = @gramos.collect{|x| x}

    sum = 0

    for i in (0...ut_al.size)
        sum += ut_al[i]*gr[i]
    end

    sum
end

def i_hc
    if (gei_diario*365) < 800
        return 1

    elsif (gei_diario*365) > 1200
        return 3

    else 
        return 2
    end
end
