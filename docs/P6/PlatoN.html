def porcentaje_ch
    gr = @gramos.collect{|x| x}
    p_al = @alimentos.collect{|x| x.kcal_ch}
    total = @alimentos.collect{|x| x.ve} 

    kcal_total = 0
    ch_total = 0

    for i in (0...total.size)
        kcal_total += (total[i]*gr[i])
        ch_total += (p_al[i]*gr[i])
    end
    @porc_ch = ch_total*100/kcal_total
end
